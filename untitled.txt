                <p class="text-[10px] text-red-600 font-bold uppercase tracking-widest">Upload New Video</p>
                <input type="text" id="title" placeholder="Video Title" class="w-full p-3 bg-black border border-gray-800 rounded-lg text-sm outline-none focus:border-red-600" required>
                <input type="url" id="link" placeholder="Video Link (YouTube/Drive)" class="w-full p-3 bg-black border border-gray-800 rounded-lg text-sm outline-none focus:border-red-600" required>

                    <label class="block text-[10px] text-gray-500 mb-1">VIDEO THUMBNAIL (GALLERY)</label>
                    <input type="file" id="thumbFile" accept="image/*" class="w-full text-xs text-gray-400 file:mr-4 file:py-2 file:px-4 file:rounded-full file:border-0 file:text-xs file:font-semibold file:bg-gray-700 file:text-white">

    <script>
        let isAdmin = false;
        const PWD = "myloveammu092";
        
        let db = JSON.parse(localStorage.getItem('yash_v3_db')) || {
            logo: 'https://via.placeholder.com/100',
            whatsapp: '910000000000',
            categories: ['birthday', 'marriage', 'trending'],
            videos: []
        };

        function adminLogin() {
            if(!isAdmin) {
                if(prompt("Admin Password?") === PWD) {
                    isAdmin = true;
                    document.getElementById('adminBtn').innerText = "OPEN PANEL";
                    render();
                }
            } else {
                updateCatDropdown();
                document.getElementById('waNumberInput').value = db.whatsapp;
                document.getElementById('modal').classList.remove('hidden');
            }
        }

        function closeModal() { document.getElementById('modal').classList.add('hidden'); }

        // Gallery se Image uthane ka function
        function getBase64(file) {
            return new Promise((resolve, reject) => {
                const reader = new FileReader();
                reader.readAsDataURL(file);
                reader.onload = () => resolve(reader.result);
                reader.onerror = error => reject(error);
            });
        }

        async function saveSettings() {
            const logoInput = document.getElementById('logoFile');
            if(logoInput.files[0]) {
                db.logo = await getBase64(logoInput.files[0]);
            }
            db.whatsapp = document.getElementById('waNumberInput').value;
            save(); render(); alert("Settings Updated!");
        }

        function addNewCategory() {
            let n = prompt("New Folder Name:");
            if(n) { db.categories.push(n.toLowerCase()); save(); updateCatDropdown(); render(); }
        }

        function updateCatDropdown() {
            document.getElementById('cat').innerHTML = db.categories.map(c => `<option value="${c}">${c.toUpperCase()}</option>`).join('');
        }

        function render() {
            document.getElementById('display-logo').style.backgroundImage = `url('${db.logo}')`;
            const main = document.getElementById('main-content');
            main.innerHTML = '';

            db.categories.forEach(cat => {
                let vids = db.videos.filter(v => v.cat === cat);
                if(vids.length === 0 && !isAdmin) return;

                let section = `
                    <section>
                        <h2 class="text-xs font-bold tracking-[0.2em] text-gray-600 mb-6 uppercase">● ${cat}</h2>
                        <div class="grid grid-cols-1 md:grid-cols-2 gap-6">
                            ${vids.map((v, i) => `
                                <div class="bg-[#0a0a0a] rounded-xl overflow-hidden border border-gray-900 card-shadow">
                                    <img src="${v.thumb}" class="w-full aspect-video object-cover">
                                    <div class="p-5">
                                        <div class="flex justify-between items-center mb-4">
                                            <h4 class="font-bold text-sm uppercase">${v.title}</h4>
                                            <span class="text-red-600 font-bold">₹${v.price}</span>
                                        </div>
                                        <div class="grid grid-cols-2 gap-2">
                                            <a href="${v.link}" target="_blank" class="text-center border border-gray-800 py-2 rounded text-[10px] font-bold">PREVIEW</a>
                                            <a href="https://wa.me/${db.whatsapp}?text=Hi, I want to book: ${v.title}" target="_blank" class="text-center bg-green-600 py-2 rounded text-[10px] font-bold">BOOK NOW</a>
                                        </div>
                                        ${isAdmin ? `<button onclick="deleteVid('${v.id}')" class="mt-4 text-[9px] text-gray-600 underline uppercase">Delete</button>` : ''}
                                    </div>
                                </div>
                            `).join('')}
                        </div>
                    </section>`;
                main.innerHTML += section;
            });
        }

        document.getElementById('uploadForm').onsubmit = async (e) => {
            e.preventDefault();
            let thumbData = 'https://via.placeholder.com/400x225/111/ff0000?text=Yash+Editz';
            const tFile = document.getElementById('thumbFile').files[0];
            
            if(tFile) { thumbData = await getBase64(tFile); }

            db.videos.push({
                id: Date.now().toString(),
                title: document.getElementById('title').value,
                link: document.getElementById('link').value,
                price: document.getElementById('price').value,
                cat: document.getElementById('cat').value,
                thumb: thumbData
            });
            save(); render(); closeModal(); e.target.reset();
        };

        function deleteVid(id) { if(confirm('Delete?')) { db.videos = db.videos.filter(v => v.id !== id); save(); render(); } }
        function save() { localStorage.setItem('yash_v3_db', JSON.stringify(db)); }
        render();
    </script>
